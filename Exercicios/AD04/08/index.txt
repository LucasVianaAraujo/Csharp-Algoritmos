namespace cSharp01;

class Program
{
    static void Main(string[] args)
    {
        ApresentarPrograma();

        Calculo();
    }

    public static void ApresentarPrograma()
    {
        Console.WriteLine("-- PROGRAMA DA NOTA --");
    }

    public static void Calculo()
    {
        Console.WriteLine("Informe a primeira nota:");
        double n1 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a segunda nota:");
        double n2 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a terceira nota:");
        double n3 = Convert.ToDouble(Console.ReadLine());

        double CalculoMedia = (n1 + n2 + n3) / 3;

        if (CalculoMedia >= 8)
        {
            Console.WriteLine("Situação: Aprovado com sucesso");
        }

        else if (CalculoMedia >= 6 && CalculoMedia < 8)
        {
            Console.WriteLine("Situação: Aprovado");
        }

        else if (CalculoMedia >= 3 && CalculoMedia < 6)
        {
            Console.WriteLine("Situação: Recuperação");
        }

        else if (CalculoMedia < 3)
        {
            Console.WriteLine("Situação: Reprovado");
        }

        else if (CalculoMedia == 0)
        {
            Console.WriteLine("Situação: Desistente");
        }

    }
}

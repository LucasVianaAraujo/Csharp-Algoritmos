namespace cSharp01;

class Program
{
    static void Main(string[] args)
    {
        ApresentarPrograma();

        Calculo();
    }

    public static void ApresentarPrograma()
    {
        Console.WriteLine("-- PROGRAMA DO ORÇAMENTO --");
    }

    public static void Calculo()
    {
        Console.WriteLine("Informe o seu orçamento mensal:");
        double OrcamentoMensal = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe o gasto máximo mensal:");
        double GastoMaximo = Convert.ToDouble(Console.ReadLine());

        if (OrcamentoMensal >= GastoMaximo)
        {
            Console.WriteLine("Você está dentro do orçamento!");
        }

        else
        {
            Console.WriteLine("Você está fora do orçamento! Não gaste mais!");
        }

    }
}
